let currentUserId = null;
let myPresenceRef = null;

function getUserName() {
    let name = localStorage.getItem("anonName");
    if (!name) {
        name = "Ẩn danh " + Math.floor(Math.random() * 10000);
        localStorage.setItem("anonName", name);

function escapeHtml(text) { 
    if (!text) return "";
    const div = document.createElement('div'); 
    div.textContent = text; 
    return div.innerHTML; 
}

function formatTime(ts) {
    if (!ts) return "";
    const diff = Math.floor((Date.now() - ts) / 1000);
    if (diff < 60) return "Vừa xong";
    if (diff < 3600) return Math.floor(diff / 60) + " phút trước";
    if (diff < 86400) return Math.floor(diff / 3600) + " giờ trước";
    return new Date(ts).toLocaleDateString('vi-VN');
}

function isToday(ts) { 
    return new Date(ts).toDateString() === new Date().toDateString(); 
}

function initPresence() {
    currentUserId = getUserName();
    myPresenceRef = presenceRef.child(currentUserId);
    
    onlineRef.on("value", (snap) => {
        if (snap.val()) {
            myPresenceRef.set({
                online: true,
                lastSeen: Date.now(),
                name: currentUserId
            });
            myPresenceRef.onDisconnect().remove();
        }
    });
    
    presenceRef.on("value", (snap) => {
        let onlineUsers = 0;
        if (snap.exists()) {
            const data = snap.val();
            for (let key in data) {
                if (data[key] && data[key].online !== false) {
                    onlineUsers++;
                }
            }
        }
        document.getElementById("onlineCount").innerText = onlineUsers;
    });
}

function showAnnouncement(message, timestamp, announcementId) {
    const snoozeUntil = localStorage.getItem("announcement_snooze_until");
    if (snoozeUntil && parseInt(snoozeUntil) > Date.now()) return;
    const closedId = localStorage.getItem("closed_announcement_id");
    if (closedId && closedId === announcementId) return;
    const box = document.getElementById("announcementBox");
    const content = document.getElementById("announcementContent");
    const timeSpan = document.getElementById("announcementTime");
    content.innerHTML = message;
    timeSpan.innerHTML = "🕒 " + new Date(timestamp).toLocaleString('vi-VN');
    box.style.display = "block";
    box.setAttribute("data-current-id", announcementId);
}

function closeAnnouncement() {
    const box = document.getElementById("announcementBox");
    const currentId = box.getAttribute("data-current-id");
    if (currentId) localStorage.setItem("closed_announcement_id", currentId);
    box.style.display = "none";
}

function reportPost(postId) {
    const reason = prompt("📝 Nhập lý do báo cáo bài viết này:");
    if (!reason || reason.trim() === "") {

async function postData() {
    const text = document.getElementById("content").value.trim();
    if (!text || !document.getElementById("agreeRule").checked) {
        alert("Vui lòng nhập nội dung và đồng ý nội quy!"); 
        return;
    }
    
    const btn = document.querySelector(".btn-post");
    const originalText = btn.innerText;
    btn.innerText = "⏳ AI đang kiểm tra...";
    btn.disabled = true;
    
    try {
        const response = await fetch(AI_API_URL, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ content: text })
        });
        const data = await response.json();
        
        if (data.result === "UNSAFE") {
            alert("❌ Nội dung vi phạm quy định!");
            btn.innerText = originalText;
            btn.disabled = false;
            return;
        }
        
        if (data.result === "REVIEW") {
            await pendingPostsRef.push({ 
                content: text, timestamp: Date.now(), authorName: getUserName(),
                status: "pending", ai_review: "REVIEW"
            });
            alert("⚠️ Bài viết đang được kiểm duyệt.");
        } else if (data.result === "SAFE") {
            await postsRef.push({ 
                content: text, timestamp: Date.now(), likes: 0, likedBy: {}
            });
            alert("✅ Bài viết đã được đăng!");
        }
        
        document.getElementById("content").value = "";
        document.getElementById("realPost").style.display = 'none';
        document.getElementById("agreeRule").checked = false;
    } catch (error) {
        console.error("Lỗi:", error);
        await pendingPostsRef.push({ 
            content: text, timestamp: Date.now(), authorName: getUserName(),
            status: "pending", ai_error: true
        });
        alert("Hệ thống bận, bài viết đã gửi admin duyệt!");
    } finally {
        btn.innerText = originalText;
        btn.disabled = false;
    }
}

function likePost(postId) {
    const userId = getUserName();
    const postRef = db.ref(`confessions/${postId}`);
    postRef.transaction(post => {
        if (!post) return post;
        if (!post.likedBy) post.likedBy = {};
        if (post.likedBy[userId]) {
            post.likes = (post.likes || 0) - 1;
            delete post.likedBy[userId];
        } else {
            post.likes = (post.likes || 0) + 1;
            post.likedBy[userId] = true;
        }
        return post;
    }).catch(err => console.error("Lỗi like:", err));
}

function toggleComment(id) {
    const section = document.getElementById(`comment-section-${id}`);
    if (!section) return;
    const isHidden = section.style.display === 'none' || section.style.display === '';
    section.style.display = isHidden ? 'block' : 'none';
    if (isHidden && !section.hasAttribute('data-listener')) {
        attachCommentsListener(id);
        section.setAttribute('data-listener', 'true');
    }
}

function attachCommentsListener(postId) {
    const listDiv = document.getElementById(`comment-list-${postId}`);
    const countSpan = document.getElementById(`comment-count-${postId}`);
    
    db.ref(`confessions/${postId}/comments`).on('value', snap => {
        const comments = snap.val();
        let count = 0;
        if (comments) for (let k in comments) count++;
        if (countSpan) {
            if (count === 0) countSpan.innerText = "Chưa có bình luận";
            else if (count === 1) countSpan.innerText = "1 bình luận";
            else countSpan.innerText = count + " bình luận";
        }
        if (listDiv) {
            listDiv.innerHTML = '';
            if (comments) listDiv.appendChild(renderCommentTree(postId, comments));
        }
    });
}

function renderCommentTree(postId, commentsObj) {
    const fragment = document.createDocumentFragment();
    for (let cmtId in commentsObj) {
        const cmt = commentsObj[cmtId];
        const div = document.createElement('div');
        div.className = 'comment-node';
        div.innerHTML = `
            <div class="comment-main">
                <div class="avatar" style="width:32px; height:32px; font-size:14px;">🙂</div>
                <div class="comment-bubble">
                    <div class="comment-name">${escapeHtml(cmt.name)}</div>
                    <div class="comment-text">${escapeHtml(cmt.text)}</div>
                    <div class="comment-actions"><span onclick="showReplyBox('${cmtId}')">Trả lời</span></div>
                </div>
            </div>
            <div class="reply-input-box" id="reply-box-${cmtId}">
                <input type="text" placeholder="Viết trả lời..." onkeypress="if(event.key==='Enter') addReply('${postId}', '${cmtId}', this)">
            </div>
            <div class="reply-children" id="replies-${cmtId}"></div>
        `;
        if (cmt.replies) {
            div.querySelector('.reply-children').appendChild(renderCommentTree(postId, cmt.replies));
        }
        fragment.appendChild(div);
    }
    return fragment;
}

function showReplyBox(cmtId) {
    const box = document.getElementById(`reply-box-${cmtId}`);
    if (box) {
        box.style.display = box.style.display === 'block' ? 'none' : 'block';
        if (box.style.display === 'block') box.querySelector('input').focus();
    }
}

function addComment(postId) {
    const input = document.getElementById(`comment-input-${postId}`);
    const text = input.value.trim();
    if (!text) return;
    db.ref(`confessions/${postId}/comments`).push({ 
        name: getUserName(), text, timestamp: Date.now() 
    });
    input.value = '';
}

function addReply(postId, parentId, input) {
    const text = input.value.trim();
    if (!text) return;
    db.ref(`confessions/${postId}/comments/${parentId}/replies`).push({ 
        name: getUserName(), text, timestamp: Date.now() 
    });
    input.value = '';
    const box = document.getElementById(`reply-box-${parentId}`);
    if (box) box.style.display = 'none';
}

function updatePostNumbers() {
    const cards = document.querySelectorAll('#list .card');
    const total = cards.length;
    cards.forEach((card, index) => {
        const numberSpan = card.querySelector('.post-number');
        if (numberSpan) {
            numberSpan.innerText = `#${total - index}`;
        }
    });
}

// Hàm render bài viết - KHÔNG xóa bài cũ
function renderPost(id, post) {
    // Kiểm tra nếu bài viết đã tồn tại thì không render lại
    if (document.querySelector(`.card[data-id="${id}"]`)) {
        // Cập nhật like count nếu cần
        const likeSpan = document.getElementById(`like-count-${id}`);
        if (likeSpan && likeSpan.innerText != (post.likes || 0)) {
            likeSpan.innerText = post.likes || 0;
        }
        return;
    }

// Load bài viết - GIỮ NGUYÊN bài cũ, chỉ thêm bài mới
function loadApprovedPosts() {
    postsRef.once("value", (snapshot) => {
        const posts = [];
        snapshot.forEach(child => {
            posts.push({
                id: child.key,
                ...child.val()
            });
        });
        
        posts.sort((a, b) => (b.timestamp || 0) - (a.timestamp || 0));
        
        const list = document.getElementById("list");
        list.innerHTML = '';
        
        posts.forEach(post => {
            const card = document.createElement('div');
            card.className = 'card';
            card.setAttribute('data-id', post.id);
            
            card.innerHTML = `
                <div class="post-header">
                    <span class="post-number">#?</span>
                    <span class="post-author">Ẩn danh</span>
                    <span class="post-time">${formatTime(post.timestamp)}</span>
                    ${isToday(post.timestamp) ? '<span class="today-badge">Hôm nay</span>' : ''}
                </div>
                <div class="post-content">${escapeHtml(post.content)}</div>
                <div class="post-stats">
                    <span>👍 <span id="like-count-${post.id}">${post.likes || 0}</span></span>
                    <span>💬 <span id="comment-count-${post.id}">Đang tải...</span></span>
                </div>
                <div class="post-actions">
                    <button onclick="likePost('${post.id}')">👍 Thích</button>
                    <button onclick="toggleComment('${post.id}')">💬 Bình luận</button>
                    <button onclick="reportPost('${post.id}')">❌ Báo cáo</button>
                </div>
                <div id="comment-section-${post.id}" class="comment-section">
                    <div class="comment-input-wrapper">
                        <input type="text" id="comment-input-${post.id}" placeholder="Viết bình luận..." onkeypress="if(event.key==='Enter') addComment('${post.id}')">
                    </div>
                    <div id="comment-list-${post.id}"></div>
                </div>
            `;
            
            list.appendChild(card);
            
            db.ref(`confessions/${post.id}/likes`).on('value', s => {
                const span = document.getElementById(`like-count-${post.id}`);
                if (span) span.innerText = s.val() || 0;
            });
            
            attachCommentsListener(post.id);
        });
        
        updatePostNumbers();
    });
}

// Lắng nghe bài viết mới - KHÔNG làm mất bài cũ
postsRef.on("child_added", (snapshot) => {
    if (!document.querySelector(`.card[data-id="${snapshot.key}"]`)) {
        renderPost(snapshot.key, snapshot.val());
    }
});

postsRef.on("child_changed", (snapshot) => {
    const post = snapshot.val();
    const likeSpan = document.getElementById(`like-count-${snapshot.key}`);
    if (likeSpan && likeSpan.innerText != (post.likes || 0)) {
        likeSpan.innerText = post.likes || 0;
    }
});

postsRef.on("child_removed", (snapshot) => {
    const card = document.querySelector(`.card[data-id="${snapshot.key}"]`);
    if (card) {
        card.remove();
        updatePostNumbers();
    }
});

// Chat functionality
let chatModal = document.getElementById("chatModal");
let chatMessagesDiv = document.getElementById("chatMessagesList");
let chatInput = document.getElementById("chatInput");
let sendBtn = document.getElementById("sendChatBtn");
let fabBtn = document.getElementById("chatFabBtn");
let closeChatBtn = document.getElementById("closeChatBtn");

function scrollChatBottom() {
    if (chatMessagesDiv) chatMessagesDiv.scrollTop = chatMessagesDiv.scrollHeight;
}

function sendChatMessage() {
    let text = chatInput.value.trim();
    if (!text) return;
    chatMessagesRef.push({
        name: getUserName(),
        text: text,
        timestamp: Date.now()
    }).then(() => {
        chatInput.value = "";
        scrollChatBottom();
    }).catch(err => console.error("Lỗi gửi tin nhắn:", err));
}

function renderChatMessage(snapshot) {
    const msg = snapshot.val();
    if (!msg) return;
    const currentUser = getUserName();
    const isOwn = (msg.name === currentUser);
    const bubbleDiv = document.createElement("div");
    bubbleDiv.className = `chat-bubble ${isOwn ? 'own' : 'other'}`;
    const timeStr = new Date(msg.timestamp).toLocaleTimeString('vi-VN', {hour:'2-digit', minute:'2-digit'});
    bubbleDiv.innerHTML = `
        <div class="chat-name">${escapeHtml(msg.name)}</div>
        <div>${escapeHtml(msg.text)}</div>
        <div class="chat-time">${timeStr}</div>
    `;
    chatMessagesDiv.appendChild(bubbleDiv);
    scrollChatBottom();
}

chatMessagesRef.limitToLast(100).on("child_added", renderChatMessage);

sendBtn.addEventListener("click", sendChatMessage);
chatInput.addEventListener("keypress", (e) => { if(e.key === "Enter") sendChatMessage(); });

fabBtn.addEventListener("click", () => {
    chatModal.classList.toggle("show");
    if (chatModal.classList.contains("show")) scrollChatBottom();
});
closeChatBtn.addEventListener("click", () => chatModal.classList.remove("show"));

loadApprovedPosts();
initPresence();

console.log("✅ Diễn đàn + Chat cộng đồng đã sẵn sàng!");
